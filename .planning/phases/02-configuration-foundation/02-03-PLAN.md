    ```
    public struct WinetricksVerbCache: Codable, Sendable {
        public var installedVerbs: Set<String>

        public var logFileSize: Int64?
        public var logFileModDate: Date?

        public init(installedVerbs: Set<String> = [], lastChecked: Date = .distantPast)

        /// Whether the cache may be stale based on winetricks.log file changes.
        public func isStale(currentLogSize: Int64?, currentLogModDate: Date?) -> Bool {
            // Stale if: no cache data (logFileSize/logFileModDate nil), or mtime changed, or size changed
        }
    }
    ```

    Add static methods for persistence on WinetricksVerbCache:
    - `static func load(from bottleURL: URL) -> WinetricksVerbCache?` -- reads `WinetricksCache.plist` from bottle directory (NOT inside the wine prefix, alongside `Metadata.plist`)
    - `static func save(_ cache: WinetricksVerbCache, to bottleURL: URL) throws` -- writes to same location
    - `static func winetricksLogInfo(for bottleURL: URL) -> (size: Int64?, modDate: Date?)` -- reads file attributes of `<bottleURL>/drive_c/winetricks.log` (the standard winetricks log location)

    Add verb discovery methods to Winetricks class in Whisky/Utils/Winetricks.swift:

    `static func listInstalledVerbs(for bottle: Bottle) async -> Set<String>?`:
    - Run `winetricks list-installed` as a headless Process (NOT through Terminal/AppleScript)
    - Set environment: WINEPREFIX=bottle.url.path, WINE=wine64, PATH includes WhiskyWineInstaller.binFolder and cabextract resource folder
    - Set executableURL to `/usr/bin/env` with arguments `["bash", winetricksURL.path, "list-installed"]` (winetricks is a bash script)
    - Capture stdout via Pipe, set a 30-second timeout using DispatchWorkItem cancellation or Task.sleep
    - Parse stdout: split by newline, trim whitespace, filter empty lines -> Set<String>
    - Return nil if process fails, times out, or produces no output
    - Log at debug level: "Running winetricks list-installed for bottle..."

    `static func parseWinetricksLog(for bottle: Bottle) -> Set<String>`:
    - Read `<bottle.url.path>/drive_c/winetricks.log` (this is where winetricks writes its log inside the prefix, not at bottle root)
    - Actually, the standard winetricks.log location is `$WINEPREFIX/winetricks.log` (at prefix root, not drive_c). Check both locations.
    - Parse lines: look for lines like "Executing load_<verb>" or "Installed <verb>" or just verb names (winetricks log format: each successfully installed verb appears on its own line in the log)
    - Return Set<String> of verb names found
    - This is the fallback; best-effort parsing

    `static func loadInstalledVerbs(for bottle: Bottle) async -> (verbs: Set<String>, fromCache: Bool)`:
    - Load cache from WinetricksVerbCache.load(from: bottle.url)
    - Check staleness via winetricksLogInfo
    - If cache exists and not stale, return (cache.installedVerbs, true)
    - Otherwise, try listInstalledVerbs first; if nil, fall back to parseWinetricksLog
    - Save updated cache with new verbs, logFileSize, logFileModDate
    - Return (verbs, false)

  <action>
    Update WinetricksView to display installed verb status:

    1. Add state properties:
       - `@State private var installedVerbs: Set<String> = []`
       - `@State private var isLoadingInstalledVerbs: Bool = true`
       - `@State private var verbFilter: VerbFilter = .all` (new enum: `.all`, `.installed`)

    2. Add a `VerbFilter` enum (private, inside the view or as a nested type):
       ```
       private enum VerbFilter: String, CaseIterable {
           case all
           case installed
       }
       ```

    3. Add a Picker (segmented style) above the TabView for filtering:
       ```
       Picker("Filter", selection: $verbFilter) {
           Text("winetricks.filter.all").tag(VerbFilter.all)
           Text("winetricks.filter.installed").tag(VerbFilter.installed)
       }
       .pickerStyle(.segmented)
       .padding(.horizontal)
       ```

    4. Modify the Table in each category tab to:
       - Filter verbs based on `verbFilter`: when `.installed`, only show verbs where `installedVerbs.contains(verb.name)`
       - Add a third TableColumn showing installed status: a checkmark (Image(systemName: "checkmark.circle.fill")) tinted green for installed verbs, empty for uninstalled
       - The installed column should be narrow (width: 30) and placed first

    5. On `.onAppear`, after loading verb categories (existing logic), also load installed verbs:
       ```
       Task.detached {
           let result = await Winetricks.loadInstalledVerbs(for: bottle)
           await MainActor.run {
               installedVerbs = result.verbs
               isLoadingInstalledVerbs = false
           }

           // If loaded from cache, do a background refresh
           if result.fromCache {
               if let fresh = await Winetricks.listInstalledVerbs(for: bottle) {
                   await MainActor.run {
                       installedVerbs = fresh
                   }
                   // Save updated cache
                   var cache = WinetricksVerbCache(installedVerbs: fresh, lastChecked: Date())
                   let logInfo = WinetricksVerbCache.winetricksLogInfo(for: bottle.url)
                   cache.logFileSize = logInfo.size
                   cache.logFileModDate = logInfo.modDate
                   try? WinetricksVerbCache.save(cache, to: bottle.url)
               }
           }
       }
       ```

    6. When `verbFilter == .installed` and `isLoadingInstalledVerbs == true`, show a small ProgressView in the filter area or as overlay text ("Loading installed verbs...")

    7. Show installed verb count in the "Installed" segment label: e.g., "Installed (\(installedVerbs.count))" if count > 0

    Note on localization: Add the following localization keys (English only, per project rules -- translations handled on Crowdin):
    - "winetricks.filter.all" = "All"
    - "winetricks.filter.installed" = "Installed"
    - "winetricks.table.installed" = "Installed" (column header)
    Add these to the EN Localizable.strings file.
